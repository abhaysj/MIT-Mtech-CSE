#include <stdio.h>
#include <stdlib.h>

void pigeonhole_sort(int arr[], int n, int min_val, int range_size, int* sorted_arr) {
    int pigeonholes[range_size];
    for (int i = 0; i < range_size; i++) {
        pigeonholes[i] = 0;
    }

    for (int i = 0; i < n; i++) {
        pigeonholes[arr[i] - min_val]++;
    }

    int index = 0;
    for (int i = 0; i < range_size; i++) {
        while (pigeonholes[i] > 0) {
            sorted_arr[index++] = i + min_val;
            pigeonholes[i]--;
        }
    }
}

int main() {
    int n;
    printf("Enter the number of elements: ");
    scanf("%d", &n);

    int arr[n];
    printf("Enter %d elements: ", n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int min_val = arr[0];
    int max_val = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] < min_val) {
            min_val = arr[i];
        }
        if (arr[i] > max_val) {
            max_val = arr[i];
        }
    }

    int range_size = max_val - min_val + 1;
    int sorted_arr[n];
    pigeonhole_sort(arr, n, min_val, range_size, sorted_arr);

    printf("Sorted array: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", sorted_arr[i]);
    }
    printf("\n");

    return 0;
}
